#include <iostream>
#include <algorithm>
#include <vector>
#include <string>

using namespace std;

int calc(vector<int>& container, int k) {
	int ans = 0, step;
	int i, j, n = container.size();
	for (i = 0; i < n - 1 && n - i > ans; i++) {
		step = container[i];
		if (ans == 0 && step % k == 0) ans = 1;
		for (j = i + 1; j < n; j++) {
			step += container[j];
			if (step %k == 0 && ans < j - i + 1)
				ans = j - i + 1;
		}
	}
	return ans;
}

int main() {
	int n, i, k;
	vector<int> container;
	cin >> n;
	for (i = 0; i < n; i++) {
		cin >> k;
		container.push_back(k);
	}
	cin >> k;
	cout << calc(container, k) << endl;
	return 0;
}
